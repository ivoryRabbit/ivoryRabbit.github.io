```R
library(titanic)
library(tidyr)
library(dplyr)
library(stringr)
library(ggplot2)

titanic_train %>% 
  names()
titanic_test %>% 
  names()

titanic_train %>% 
  select(-Survived) %>% 
  rbind(titanic_test) ->
  titanic_df

# Pclass
titanic_df %>% 
  select(Pclass) %>% 
  unique()

titanic_df %>%
  mutate(Pclass = as.factor(Pclass)) %>% 
  group_by(Pclass) %>% 
  summarise(
    cnt = n()
  ) %>% 
  ggplot(aes(x = Pclass, y = cnt, fill = factor(Pclass))) + 
  geom_col(width = .5, show.legend = F) + 
  ylab('인원수(명')
```
